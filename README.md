
# install ON Kali Linux :
```
   sudo apt-get update && sudo apt-get -y full-upgrade
   sudo apt install git python
   git clone https://github.com/hacker6291/number-info.git
   cd number-info
   

```

# install ON TERMUX :
```
   pkg update && upgrade
   pkg install git python
   git clone https://github.com/hacker6291/number-info.git
   cd number-info

```

# RUNNING
```
   cd number-info
   ls
   
   python3 number-info.py
```
